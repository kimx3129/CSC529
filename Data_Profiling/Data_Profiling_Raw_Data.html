    #Create a dataframe containing only missing values

    nan_patt = null_data.isnull()
    for npv in nan_patt.values:
        #Create a missing value pattern
        patt_str = ""
        for p in npv:
            if p:
                patt_str = patt_str + "|"
            else:
                patt_str = patt_str + "."

        if flag == 0:
            temp2["Pattern"] = patt_str
            temp2["Pattern_Type"] = patt_idx
            patt_idx += 1
            flag += 1
        else:
            pp = pd.DataFrame(index = range(0,1))
            pp["Pattern"] = patt_str
            if patt_str not in temp2.Pattern.values:
                pp["Pattern_Type"] = patt_idx
                patt_idx += 1
            else:
                ix = pp["Pattern"][0] == temp2.Pattern.values
                dup_patt_idx = np.where(ix == True)[0][0]
                pp["Pattern_Type"] = temp2["Pattern_Type"].iloc[dup_patt_idx]
            temp2 = temp2.append(pp)

    # Pattern and its type number
    xx = pd.DataFrame({'Count' : temp2.groupby( ["Pattern", "Pattern_Type"] ).size()}).reset_index()
