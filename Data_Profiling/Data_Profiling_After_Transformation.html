            patt_idx += 1
            flag += 1
        else:
            pp = pd.DataFrame(index = range(0,1))
            pp["Pattern"] = patt_str
            if patt_str not in temp2.Pattern.values:
                pp["Pattern_Type"] = patt_idx
                patt_idx += 1
            else:
                ix = pp["Pattern"][0] == temp2.Pattern.values
                dup_patt_idx = np.where(ix == True)[0][0]
                pp["Pattern_Type"] = temp2["Pattern_Type"].iloc[dup_patt_idx]
